```javascript
// js/timezone.js
const TIMEZONES = {
  beijing: { label: 'beijing', offset: 8, flag: '🇨🇳' },
  eastern: { label: 'eastern', offset: -4, flag: '🇺🇸' },
  gmt: { label: 'gmt', offset: 0, flag: '🌐' }
};

let currentTZ = localStorage.getItem('wc-timezone') || detectTimezone();

function detectTimezone() {
  const offset = -new Date().getTimezoneOffset() / 60;
  if (offset === 8) return 'beijing';
  if (offset >= -5 && offset <= -4) return 'eastern';
  return 'gmt';
}

function setTimezone(key) {
  currentTZ = key;
  localStorage.setItem('wc-timezone', key);
}

function convertTime(timeStr) {
  // "13:00 UTC-6" → convert to current timezone
  const match = timeStr.match(/(\d{2}):(\d{2})\s+UTC([+-]\d+)/);
  if (!match) return timeStr;
  const [, h, m, utcOff] = match;
  let date = new Date(Date.UTC(2026, 5, 11, parseInt(h) - parseInt(utcOff), parseInt(m)));
  const targetOff = TIMEZONES[currentTZ].offset;
  date.setHours(date.getHours() + targetOff);
  return date.getHours().toString().padStart(2, '0') + ':' + date.getMinutes().toString().padStart(2, '0');
}

function getUTCOffsetStr() {
  const off = TIMEZONES[currentTZ].offset;
  const sign = off >= 0 ? '+' : '';
  return `UTC${sign}${off}`;
}
```

```javascript
// js/data.js
const CDN_URL = 'https://cdn.jsdelivr.net/gh/openfootball/worldcup.json@master/2026/worldcup.json';
const CACHE_KEY = 'wc-data';
const CACHE_TIME_KEY = 'wc-data-time';
const CACHE_TTL = 3600000; // 1 hour

let worldCupData = null;

async function loadData() {
  // Try cache first
  const cached = localStorage.getItem(CACHE_KEY);
  const cacheTime = localStorage.getItem(CACHE_TIME_KEY);
  if (cached && cacheTime && (Date.now() - parseInt(cacheTime) < CACHE_TTL)) {
    worldCupData = JSON.parse(cached);
    return worldCupData;
  }

  try {
    const resp = await fetch(CDN_URL);
    if (resp.ok) {
      worldCupData = await resp.json();
      localStorage.setItem(CACHE_KEY, JSON.stringify(worldCupData));
      localStorage.setItem(CACHE_TIME_KEY, Date.now().toString());
      return worldCupData;
    }
  } catch (e) {
    console.warn('CDN load failed, trying fallback...');
  }

  // Fallback to local
  try {
    const resp = await fetch('data/worldcup.json');
    worldCupData = await resp.json();
    return worldCupData;
  } catch (e) {
    console.error('Failed to load data');
    return null;
  }
}

function getMatches() {
  if (!worldCupData) return [];
  return worldCupData.matches.map((m, i) => {
    if (!m.num) m.num = i + 1; // fill missing num by array order
    return m;
  });
}

function getGroupMatches() {
  return getMatches().filter(m => m.group && m.group.startsWith('Group '));
}

function getKnockoutMatches() {
  return getMatches().filter(m => !m.group || !m.group.startsWith('Group '));
}

function getGroups() {
  const groups = new Set();
  getGroupMatches().forEach(m => groups.add(m.group));
  return [...groups].sort();
}

function getTeams() {
  const teams = new Set();
  getMatches().forEach(m => {
    if (m.team1 && !m.team1.startsWith('W') && !m.team1.startsWith('L')) teams.add(m.team1);
    if (m.team2 && !m.team2.startsWith('W') && !m.team2.startsWith('L')) teams.add(m.team2);
  });
  return [...teams].sort();
}

// Compute standings from match results
function computeStandings(groupName) {
  const matches = getGroupMatches().filter(m => m.group === groupName);
  const teams = {};
  matches.forEach(m => {
    if (!teams[m.team1]) teams[m.team1] = newRecord();
    if (!teams[m.team2]) teams[m.team2] = newRecord();
    if (m.score1 != null && m.score2 != null) {
      const t1 = teams[m.team1], t2 = teams[m.team2];
      t1.played++; t2.played++;
      t1.gf += m.score1; t1.ga += m.score2;
      t2.gf += m.score2; t2.ga += m.score1;
      t1.gd = t1.gf - t1.ga; t2.gd = t2.gf - t2.ga;
      if (m.score1 > m.score2) { t1.won++; t2.lost++; t1.pts += 3; }
      else if (m.score1 < m.score2) { t2.won++; t1.lost++; t2.pts += 3; }
      else { t1.drawn++; t2.drawn++; t1.pts += 1; t2.pts += 1; }
    }
  });
  return Object.entries(teams)
    .map(([name, r]) => ({ name, ...r }))
    .sort((a, b) => b.pts - a.pts || b.gd - a.gd || b.gf - a.gf);
}

function newRecord() {
  return { played: 0, won: 0, drawn: 0, lost: 0, gf: 0, ga: 0, gd: 0, pts: 0 };
}
```

```javascript
// js/schedule.js
function renderSchedule(filterGroup, filterTeam) {
  const container = document.getElementById('schedule-list');
  const matches = getMatches();
  if (!matches.length) {
    container.innerHTML = `<div class="no-data">${t('noData')}</div>`;
    return;
  }

  // Filter
  let filtered = matches;
  if (filterGroup && filterGroup !== 'all') {
    filtered = filtered.filter(m => m.group === filterGroup);
  }
  if (filterTeam && filterTeam !== 'all') {
    filtered = filtered.filter(m => m.team1 === filterTeam || m.team2 === filterTeam);
  }

  // Group by date
  const byDate = {};
  filtered.forEach(m => {
    if (!byDate[m.date]) byDate[m.date] = [];
    byDate[m.date].push(m);
  });

  const today = new Date().toISOString().slice(0, 10);
  let html = '';

  for (const [date, dayMatches] of Object.entries(byDate).sort()) {
    const d = new Date(date + 'T00:00:00');
    const weekdays = ['周日', '周一', '周二', '周三', '周四', '周五', '周六'];
    const weekdaysEn = ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'];
    const wd = currentLang === 'zh' ? weekdays[d.getDay()] : weekdaysEn[d.getDay()];
    const dateStr = `${d.getMonth() + 1}月${d.getDate()}日 ${wd}`;
    const dateStrEn = `${d.toLocaleDateString('en', { month: 'short', day: 'numeric' })} ${wd}`;
    const displayDate = currentLang === 'zh' ? dateStr : dateStrEn;
    const isToday = date === today;

    html += `<div class="date-group"><div class="date-header${isToday ? ' today' : ''}">`;
    if (isToday) html += `★ ${t('today')} · `;
    html += `${displayDate}</div><div class="matches-grid">`;

    dayMatches.forEach(m => {
      const time = convertTime(m.time);
      const flag1 = getFlag(m.team1);
      const flag2 = getFlag(m.team2);
      const isGroup = m.group && m.group.startsWith('Group ');
      const stageLabel = isGroup ? m.group : t(roundKey(m.round));
      const hasScore = m.score1 != null && m.score2 != null;
      const scoreDisplay = hasScore ? `${m.score1} - ${m.score2}` : t('vs');

      html += `<div class="match-card">
        ${!isGroup ? `<span class="match-round">${stageLabel}</span>` : ''}
        <div class="match-time">${time} (${getUTCOffsetStr()}) · ${m.ground}</div>
        <div class="match-teams">
          <div class="team"><span class="flag">${flag1}</span><span class="name">${m.team1}</span></div>
          <div class="score">${scoreDisplay}</div>
          <div class="team"><span class="flag">${flag2}</span><span class="name">${m.team2}</span></div>
        </div>
        <div class="match-ground">${isGroup ? stageLabel : m.ground}</div>
      </div>`;
    });

    html += `</div></div>`;
  }

  container.innerHTML = html || `<div class="no-data">${t('noData')}</div>`;
}
```

```javascript
// js/standings.js
function renderStandings(groupName) {
  const container = document.getElementById('standings-content');
  const groups = getGroups();
  const target = groupName || groups[0] || 'Group A';
  const standings = computeStandings(target);

  let html = `<div class="filters">
    <select id="standings-group" onchange="onStandingsGroupChange()">`;
  groups.forEach(g => {
    html += `<option value="${g}" ${g === target ? 'selected' : ''}>${g}</option>`;
  });
  html += `</select></div>`;

  if (!standings.length) {
    html += `<div class="no-data">${t('noData')}</div>`;
    container.innerHTML = html;
    return;
  }

  html += `<table class="standings-table">
    <thead><tr>
      <th>${t('team')}</th><th>${t('played')}</th><th>${t('won')}</th>
      <th>${t('drawn')}</th><th>${t('lost')}</th><th>${t('gf')}</th>
      <th>${t('ga')}</th><th>${t('gd')}</th><th>${t('pts')}</th>
    </tr></thead><tbody>`;
```

```javascript
// js/knockout.js
const KO_ROUNDS = [
  'Round of 32', 'Round of 16', 'Quarter-final',
  'Semi-final', 'Match for third place', 'Final'
];

function renderKnockout() {
  const container = document.getElementById('knockout-content');
  const matches = getKnockoutMatches();

  // Group by round
  const byRound = {};
  KO_ROUNDS.forEach(r => { byRound[r] = []; });
  matches.forEach(m => {
    const r = m.round;
    if (byRound[r]) byRound[r].push(m);
  });

  // Filter empty rounds
  const activeRounds = KO_ROUNDS.filter(r => byRound[r].length > 0);

  let html = '<div class="bracket-container"><div class="bracket">';

  activeRounds.forEach(round => {
    html += `<div class="bracket-round"><h3>${t(roundKey(round))}</h3>`;
    byRound[round].forEach(m => {
      const hasScore = m.score1 != null && m.score2 != null;
      const w1 = hasScore && m.score1 > m.score2;
      const w2 = hasScore && m.score2 > m.score1;
      html += `<div class="bracket-match">
        <div class="bm-teams">
          <div class="bm-team${w1 ? ' winner' : ''}">
            <span>${getFlag(m.team1)} ${m.team1}</span>
            ${hasScore ? `<span>${m.score1}</span>` : ''}
          </div>
          <div class="bm-team${w2 ? ' winner' : ''}">
            <span>${getFlag(m.team2)} ${m.team2}</span>
            ${hasScore ? `<span>${m.score2}</span>` : ''}
          </div>
        </div>
        <div class="bm-info">${m.date} · ${convertTime(m.time)} · ${m.ground}</div>
      </div>`;
    });
    html += '</div>';
  });

  html += '</div></div>';
  container.innerHTML = html;
}
```
